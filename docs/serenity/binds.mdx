Binds are practically just shortcuts to do certain things without having to open the menu.

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs queryString="stance">
  <TabItem value="normal" label="Standing">
    | Button    | What it does |
    | :-------- | :------- |
    | **RB** + **Dpad Left** | Suicide |
  </TabItem>
  <TabItem value="crouched" label="Crouched">
    | Button    | What it does |
    | :-------- | :------- |
    | **Dpad Up** | Drops a random weapon on the floor for CanSwaps |
    | **Dpad Down** | Give you a RPG that you can use to fly |
    | **Dpad Left** | Sets your spawn to where you are standing |
  </TabItem>
</Tabs>
